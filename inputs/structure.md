  <!-- .element: class="fragment fade-left" -->
+ <div>**lambda abstractions, higher-order-functions**

  <!-- .element: class="fragment fade-left" -->
+ <div>**denotational semantics**</div>
  <!-- .element: class="fragment fade-left" -->
+ <div>**type theory**  
  the "theory of abstraction"</div>
  <!-- .element: class="fragment fade-left" -->
+ <div>**proof assistants**  
  Agda, Coq, NuPRL, Isabelle</div>
  <!-- .element: class="fragment fade-left" -->
